####  allow .htaccess authentication ####



step 1 : Create a password file with htpasswd

        #  htpasswd -c /etc/feathersoft-tr2/ libin ..............................( user name )

step 2 :Now, you need to allow the apache user to read the .htpasswd file

       # chown apache:apache /app/.htpasswd

       # chmod 0660 /app/.htpasswd

step 3 :Configure Apache password authentication

       # cd /var/www/html/hello.local/tr/

       # vim /var/www/html/hello.local/tr/.htaccess

       AuthType Basic
       AuthName "Restricted Content"
       AuthUserFile /app/.htpasswd
       Require valid-user

step 4 :restart service

       # apachectl restart
